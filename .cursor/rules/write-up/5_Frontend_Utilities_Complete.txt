## 1. INTRODUCTION TO FRONTEND UTILITIES

The frontend utilities in Jibu Console provide a set of tools and abstractions that simplify common tasks, promote code reuse, and ensure consistency across the application. These utilities include API clients for interacting with the backend, context providers for state management, authentication utilities, and various helper functions.

The frontend utilities are organized into several categories:

1. API Utilities: Clients for interacting with the backend APIs
2. Context Providers: React contexts for global state management
3. Authentication Utilities: Utilities for handling user authentication
4. Helper Functions: General-purpose utility functions

This document focuses on the API utilities, which provide type-safe clients for interacting with the backend APIs, and context providers for state management.

## 2. API UTILITIES

### 2.1. Base API Client

The base API client provides a foundation for all other API clients in the application. It handles common functionality such as making HTTP requests, handling authentication, and caching responses.

File: apps/frontend/src/utils/api.ts

The base API client provides methods for making HTTP requests to the backend API, including GET, POST, PUT, and DELETE operations. It also handles authentication by including the appropriate headers in each request.

Key Features:
- Type-safe request and response handling
- Authentication token management
- Request caching with configurable TTL (Time To Live)
- Error handling and response normalization
- Support for pagination and filtering

Implementation (Simplified):

```typescript
export interface PaginationParams {
  page?: number;
  limit?: number;
  sortBy?: string;
  sortOrder?: 'asc' | 'desc';
  search?: string;
  [key: string]: any;
}

export interface PaginatedResponse<T> {
  items: T[];
  meta: {
    total: number;
    page: number;
    limit: number;
    totalPages: number;
  };
}

export interface RequestOptions {
  headers?: Record<string, string>;
  cache?: boolean;
  cacheTTL?: number;
}

export class ApiClient {
  private baseUrl: string;
  private headers: Record<string, string>;
  private cache: Map<string, { data: any; timestamp: number }> = new Map();
  private cacheTTL: number = 5000; // 5 seconds

  constructor(baseUrl: string) {
    this.baseUrl = baseUrl;
    this.headers = {
      'Content-Type': 'application/json',
      'Accept': 'application/json',
    };
  }

  setAuthToken(token: string): void {
    this.headers['Authorization'] = `Bearer ${token}`;
  }

  getBaseUrl(): string {
    return this.baseUrl;
  }

  getHeaders(): Record<string, string> {
    return { ...this.headers };
  }

  private getCacheKey(endpoint: string, params?: Record<string, any>): string {
    return `${endpoint}${params ? `?${new URLSearchParams(params as any).toString()}` : ''}`;
  }

  private isCacheValid(cacheKey: string): boolean {
    const cachedData = this.cache.get(cacheKey);
    if (!cachedData) return false;

    const now = Date.now();
    return now - cachedData.timestamp < this.cacheTTL;
  }

  async get<T>(
    endpoint: string,
    params?: Record<string, any>,
    options?: RequestOptions
  ): Promise<T> {
    const cacheKey = this.getCacheKey(endpoint, params);

    if (options?.cache && this.isCacheValid(cacheKey)) {
      return this.cache.get(cacheKey)!.data;
    }

    const url = new URL(`${this.baseUrl}${endpoint}`);
    if (params) {
      Object.entries(params).forEach(([key, value]) => {
        if (value !== undefined) {
          url.searchParams.append(key, String(value));
        }
      });
    }

    const response = await fetch(url.toString(), {
      method: 'GET',
      headers: { ...this.headers, ...options?.headers },
      credentials: 'include',
    });

    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`);
    }

    const data = await response.json();

    if (options?.cache) {
      this.cache.set(cacheKey, {
        data,
        timestamp: Date.now(),
      });
    }

    return data;
  }

  async post<T>(
    endpoint: string,
    body: any,
    options?: RequestOptions
  ): Promise<T> {
    const url = `${this.baseUrl}${endpoint}`;
    const headers = { ...this.headers, ...options?.headers };
    const isFormData = body instanceof FormData;

    // Remove Content-Type header if FormData is used (browser will set it with boundary)
    if (isFormData && headers['Content-Type'] === 'application/json') {
      delete headers['Content-Type'];
    }

    const response = await fetch(url, {
      method: 'POST',
      headers,
      body: isFormData ? body : JSON.stringify(body),
      credentials: 'include',
    });

    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`);
    }

    return await response.json();
  }

  async put<T>(
    endpoint: string,
    body: any,
    options?: RequestOptions
  ): Promise<T> {
    const url = `${this.baseUrl}${endpoint}`;
    const response = await fetch(url, {
      method: 'PUT',
      headers: { ...this.headers, ...options?.headers },
      body: JSON.stringify(body),
      credentials: 'include',
    });

    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`);
    }

    return await response.json();
  }

  async delete<T>(
    endpoint: string,
    options?: RequestOptions
  ): Promise<T> {
    const url = `${this.baseUrl}${endpoint}`;
    const response = await fetch(url, {
      method: 'DELETE',
      headers: { ...this.headers, ...options?.headers },
      credentials: 'include',
    });

    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`);
    }

    if (response.status === 204) {
      return {} as T;
    }

    return await response.json();
  }
}
```

```typescript
      { cache: true }

  }
```

```typescript
    return this.apiClient.post<ChatMessage>(
      `${this.baseEndpoint}/${id}/messages`,
      sendMessageDto
    );
  }

    const baseUrl = this.apiClient.getBaseUrl();
    const headers = this.apiClient.getHeaders();
    const params = new URLSearchParams();

    return new EventSource(url);
```

```typescript
import { ApiClient, PaginatedResponse, PaginationParams } from './api';

export interface File {
  id: string;
  fileName: string;
  fileType: string;
  fileSize: number;
  url: string;
  organizationId: string;
  userId: string;
  metadata?: Record<string, any>;

export interface UploadFileDto {
  file: Blob;
  fileName?: string;
  metadata?: Record<string, any>;
}

export interface SupportedFileType {
  extension: string;
  mimeType: string;
  maxSize: number;
}

export class FileApi {
  private apiClient: ApiClient;
  private baseEndpoint = '/api/v1/files';

  constructor(apiClient: ApiClient) {
    this.apiClient = apiClient;
  }

  async getFiles(params?: PaginationParams): Promise<PaginatedResponse<File>> {
    return this.apiClient.get<PaginatedResponse<File>>(
      this.baseEndpoint,
      params,
      { cache: true }
    );
  }

  async getFile(id: string): Promise<File> {
    return this.apiClient.get<File>(`${this.baseEndpoint}/${id}`, undefined, {
      cache: true,
    });
  }

  async uploadFile(uploadFileDto: UploadFileDto): Promise<File> {
    const formData = new FormData();
    formData.append('file', uploadFileDto.file);
    
    if (uploadFileDto.fileName) {
      formData.append('fileName', uploadFileDto.fileName);
    }
    
    if (uploadFileDto.metadata) {
      formData.append('metadata', JSON.stringify(uploadFileDto.metadata));
    }
    
    return this.apiClient.post<File>(
      this.baseEndpoint,
      formData,
      { headers: { 'Content-Type': 'multipart/form-data' } }
    );
  }

  async deleteFile(id: string): Promise<void> {
    return this.apiClient.delete<void>(`${this.baseEndpoint}/${id}`);
  }

  async downloadFile(id: string): Promise<Blob> {
    const response = await fetch(`${this.apiClient.getBaseUrl()}${this.baseEndpoint}/${id}/download`, {
      method: 'GET',
      headers: this.apiClient.getHeaders(),
      credentials: 'include',
    });
    
    if (!response.ok) {
      throw new Error('Failed to download file');
    }
    
    return response.blob();
  }

  async getSupportedFileTypes(): Promise<SupportedFileType[]> {
    return this.apiClient.get<SupportedFileType[]>(
      `${this.baseEndpoint}/supported-types`,
      undefined,
      { cache: true }
    );
  }
}
```

```typescript
  const [isInitialized, setIsInitialized] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    const initializeApi = async () => {
      try {
        const baseUrl = process.env.NEXT_PUBLIC_API_URL || '';
        const client = new ApiClient(baseUrl);
        
        // Set auth token if user is authenticated
        if (session?.access_token) {
          client.setAuthToken(session.access_token);
        }
        
        setApiClient(client);
        setAgentApi(new AgentApi(client));
        setAssistantsApi(new AssistantsApi(client));
        setChatApi(new ChatApi(client));
        setKnowledgeBaseApi(new KnowledgeBaseApi(client));
        setFileApi(new FileApi(client));
        setIsInitialized(true);
      } catch (err) {
        setError(err instanceof Error ? err : new Error('Failed to initialize API'));
      }
    };

    initializeApi();
  }, [session]);

  return (
    <ApiContext.Provider
      value={{

        agentApi,
        assistantsApi,

        knowledgeBaseApi,
```

### 3.2. Organization Context Provider

The Organization Context Provider provides access to organization data throughout the application.

File: apps/frontend/src/contexts/OrganizationContext.tsx

The Organization Context Provider manages the current organization and provides methods for switching between organizations.

Key Features:
- Provides access to the current organization
- Provides a list of available organizations
- Handles organization switching
- Persists the selected organization in local storage

Implementation (Simplified):

```typescript
import React, { createContext, useContext, useState, useEffect } from 'react';
import { useApi } from './ApiContext';
import { useSupabase } from './SupabaseContext';

export interface Organization {
  id: string;
  name: string;
  slug: string;
  logo?: string;
  createdAt: string;
  updatedAt: string;
}

interface OrganizationContextType {
  currentOrganization: Organization | null;
  organizations: Organization[];
  isLoading: boolean;
  error: Error | null;
  switchOrganization: (organizationId: string) => Promise<void>;
}

const OrganizationContext = createContext<OrganizationContextType>({
  currentOrganization: null,
  organizations: [],
  isLoading: true,
  error: null,
  switchOrganization: async () => {},
});

export const useOrganization = () => useContext(OrganizationContext);

export const OrganizationProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const { apiClient, isInitialized } = useApi();
  const { session } = useSupabase();
  const [currentOrganization, setCurrentOrganization] = useState<Organization | null>(null);
  const [organizations, setOrganizations] = useState<Organization[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    const loadOrganizations = async () => {
      if (!isInitialized || !apiClient || !session) {
        return;
      }

      try {
        setIsLoading(true);
        
        // Fetch organizations
        const response = await apiClient.get<{ items: Organization[] }>('/api/v1/organizations');
        const orgs = response.items || [];
        setOrganizations(orgs);
        
        // Get stored organization ID from local storage
        const storedOrgId = localStorage.getItem('currentOrganizationId');
        
        // Set current organization
        if (storedOrgId && orgs.some(org => org.id === storedOrgId)) {
          setCurrentOrganization(orgs.find(org => org.id === storedOrgId) || null);
        } else if (orgs.length > 0) {
          setCurrentOrganization(orgs[0]);
          localStorage.setItem('currentOrganizationId', orgs[0].id);
        }
      } catch (err) {
        setError(err instanceof Error ? err : new Error('Failed to load organizations'));
      } finally {
        setIsLoading(false);
      }
    };

    loadOrganizations();
  }, [isInitialized, apiClient, session]);

  const switchOrganization = async (organizationId: string) => {
    const org = organizations.find(o => o.id === organizationId);
    
    if (org) {
      setCurrentOrganization(org);
      localStorage.setItem('currentOrganizationId', org.id);
    } else {
      throw new Error('Organization not found');
    }
  };

  return (
    <OrganizationContext.Provider
      value={{
        currentOrganization,
        organizations,
        isLoading,
        error,
        switchOrganization,
      }}
    >
      {children}
    </OrganizationContext.Provider>
  );
};
```

### 3.3. Chat Context Provider

The Chat Context Provider provides access to chat data and functionality throughout the application.

File: apps/frontend/src/contexts/ChatContext.tsx

The Chat Context Provider manages the current chat, chat messages, and provides methods for sending messages and handling chat responses. It integrates with the Redis-based chat history system to efficiently retrieve and manage chat messages.

Key Features:
- Provides access to the current chat and its messages
- Handles sending messages and receiving responses
- Manages loading and error states
- Provides real-time updates using Server-Sent Events (SSE)

Implementation (Simplified):

```typescript
import React, { createContext, useContext, useState, useEffect } from 'react';
import { useApi } from './ApiContext';
import { Chat, ChatMessage, SendMessageDto } from '../utils/chatApi';

interface ChatContextType {
  currentChat: Chat | null;
  messages: ChatMessage[];
  isLoading: boolean;
  isSending: boolean;
  error: Error | null;
  setCurrentChat: (chat: Chat | null) => void;
  sendMessage: (content: string, metadata?: Record<string, any>) => Promise<void>;
  clearMessages: () => void;
}

const ChatContext = createContext<ChatContextType>({
  currentChat: null,
  messages: [],
  isLoading: false,
  isSending: false,
  error: null,
  setCurrentChat: () => {},
  sendMessage: async () => {},
  clearMessages: () => {},
});

export const useChat = () => useContext(ChatContext);

export const ChatProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const { chatApi, isInitialized } = useApi();
  const [currentChat, setCurrentChat] = useState<Chat | null>(null);
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [isSending, setIsSending] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const [eventSource, setEventSource] = useState<EventSource | null>(null);

  // Load messages when current chat changes
  useEffect(() => {
    if (!isInitialized || !chatApi || !currentChat) {
      setMessages([]);
      return;
    }

    const loadMessages = async () => {
      try {
        setIsLoading(true);
        setError(null);
        
        const response = await chatApi.getChatMessages(currentChat.id);
        setMessages(response.items);
      } catch (err) {
        setError(err instanceof Error ? err : new Error('Failed to load messages'));
      } finally {
        setIsLoading(false);
      }
    };

    loadMessages();

    // Clean up event source when chat changes
    return () => {
      if (eventSource) {
        eventSource.close();
        setEventSource(null);
      }
    };
  }, [isInitialized, chatApi, currentChat]);

  const sendMessage = async (content: string, metadata?: Record<string, any>) => {
    if (!chatApi || !currentChat) {
      throw new Error('Chat API or current chat not initialized');
    }

    try {
      setIsSending(true);
      setError(null);
      
      // Send user message
      const userMessage = await chatApi.sendMessage(currentChat.id, { content, metadata });
      setMessages(prev => [...prev, userMessage]);
      
      // Close previous event source if exists
      if (eventSource) {
        eventSource.close();
      }
      
      // Stream assistant response
      const sse = chatApi.streamChatResponse(currentChat.id, { content, metadata });
      setEventSource(sse);
      
      // Handle streaming response
      let assistantMessageId = '';
      let assistantMessageContent = '';
      
      sse.onmessage = (event) => {
        try {
          const data = JSON.parse(event.data);
          
          if (data.type === 'message_start') {
            assistantMessageId = data.messageId;
            assistantMessageContent = '';
            
            // Add initial empty message
            const assistantMessage: ChatMessage = {
              id: assistantMessageId,
              chatId: currentChat.id,
              role: 'assistant',
              content: '',
              createdAt: new Date().toISOString(),
            };
            
            setMessages(prev => [...prev, assistantMessage]);
          } else if (data.type === 'message_chunk') {
            assistantMessageContent += data.chunk;
            
            // Update message with new content
            setMessages(prev => {
              const updated = [...prev];
              const index = updated.findIndex(m => m.id === assistantMessageId);
              
              if (index !== -1) {
                updated[index] = {
                  ...updated[index],
                  content: assistantMessageContent,
                };
              }
              
              return updated;
            });
          } else if (data.type === 'message_end') {
            // Message is complete, close event source
            sse.close();
            setEventSource(null);
          }
        } catch (err) {
          console.error('Error parsing SSE message:', err);
        }
      };
      
      sse.onerror = (err) => {
        console.error('SSE error:', err);
        sse.close();
        setEventSource(null);
        setError(new Error('Error receiving chat response'));
      };
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Failed to send message'));
    } finally {
      setIsSending(false);
    }
  };

  const clearMessages = () => {
    setMessages([]);
  };

  return (
    <ChatContext.Provider
      value={{
        currentChat,
        messages,
        isLoading,
        isSending,
        error,
        setCurrentChat,
        sendMessage,
        clearMessages,
      }}
    >
      {children}
    </ChatContext.Provider>
  );
};
```
