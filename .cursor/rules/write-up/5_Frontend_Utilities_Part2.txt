1.1. Chat API Client
------------------

The Chat API client provides methods for interacting with the chat-related endpoints of the backend API.

File: apps/frontend/src/utils/chatApi.ts

The Chat API client extends the base API client and provides methods for creating, retrieving, updating, and deleting chats, as well as sending messages and retrieving chat history.

Key Methods:
- getChats: Retrieves a list of chats with pagination and filtering
- getChat: Retrieves a single chat by ID
- createChat: Creates a new chat
- updateChat: Updates an existing chat
- deleteChat: Deletes a chat
- sendMessage: Sends a message in a chat
- getMessages: Retrieves messages in a chat
- streamMessages: Streams messages in a chat in real-time

Implementation:

```typescript
import { ApiClient, PaginatedResponse, PaginationParams } from './api';

export interface Chat {
  id: string;
  assistantId: string;
  userId: string;
  organizationId: string;
  sessionId: string;
  title?: string;
  status: ChatStatus;
  metadata?: Record<string, any>;
  createdAt: string;
  updatedAt: string;
}

export enum ChatStatus {
  ACTIVE = 'active',
  ARCHIVED = 'archived',
  DELETED = 'deleted',
}

export interface ChatMessage {
  id: string;
  chatId: string;
  content: string;
  role: ChatMessageRole;
  createdAt: string;
  metadata?: Record<string, any>;
}

export enum ChatMessageRole {
  USER = 'user',
  ASSISTANT = 'assistant',
  SYSTEM = 'system',
}

export interface CreateChatDto {
  assistantId: string;
  organizationId: string;
  sessionId?: string;
  title?: string;
  metadata?: Record<string, any>;
}

export interface UpdateChatDto {
  title?: string;
  status?: ChatStatus;
  metadata?: Record<string, any>;
}

export interface SendMessageDto {
  content: string;
  role: ChatMessageRole;
  metadata?: Record<string, any>;
}

export interface GetMessagesOptions {
  limit?: number;
  before?: string;
}

export class ChatApi {
  private apiClient: ApiClient;

  constructor(apiClient: ApiClient) {
    this.apiClient = apiClient;
  }

  async getChats(params?: PaginationParams): Promise<PaginatedResponse<Chat>> {
    return this.apiClient.get<PaginatedResponse<Chat>>('/v1/chats', params, { cache: true });
  }

  async getChat(id: string): Promise<Chat> {
    return this.apiClient.get<Chat>(`/v1/chats/${id}`, undefined, { cache: true });
  }

  async createChat(createChatDto: CreateChatDto): Promise<Chat> {
    const chat = await this.apiClient.post<Chat>('/v1/chats', createChatDto);
    this.apiClient.invalidateCache('/v1/chats');
    return chat;
  }

  async updateChat(id: string, updateChatDto: UpdateChatDto): Promise<Chat> {
    const chat = await this.apiClient.put<Chat>(`/v1/chats/${id}`, updateChatDto);
    this.apiClient.invalidateCache('/v1/chats');
    this.apiClient.invalidateCache(`/v1/chats/${id}`);
    return chat;
  }

  async deleteChat(id: string): Promise<void> {
    await this.apiClient.delete<void>(`/v1/chats/${id}`);
    this.apiClient.invalidateCache('/v1/chats');
    this.apiClient.invalidateCache(`/v1/chats/${id}`);
  }

  async sendMessage(id: string, sendMessageDto: SendMessageDto): Promise<ChatMessage> {
    return this.apiClient.post<ChatMessage>(`/v1/chats/${id}/messages`, sendMessageDto);
  }

  async getMessages(id: string, options?: GetMessagesOptions): Promise<ChatMessage[]> {
    return this.apiClient.get<ChatMessage[]>(`/v1/chats/${id}/messages`, options, { cache: true });
  }

  streamMessages(id: string): EventSource {
    const url = new URL(`${this.apiClient.getBaseUrl()}/v1/chats/${id}/stream`);
    return new EventSource(url.toString(), { withCredentials: true });
  }
}
```

1.2. Knowledge Base API Client
----------------------------

The Knowledge Base API client provides methods for interacting with the knowledge base-related endpoints of the backend API.

File: apps/frontend/src/utils/knowledgebaseApi.ts

The Knowledge Base API client extends the base API client and provides methods for creating, retrieving, updating, and deleting knowledge bases, as well as adding documents, listing documents, and querying knowledge bases.

Key Methods:
- getKnowledgeBases: Retrieves a list of knowledge bases with pagination and filtering
- getKnowledgeBase: Retrieves a single knowledge base by ID
- createKnowledgeBase: Creates a new knowledge base
- updateKnowledgeBase: Updates an existing knowledge base
- deleteKnowledgeBase: Deletes a knowledge base
- addDocuments: Adds documents to a knowledge base
- listDocuments: Lists documents in a knowledge base
- removeDocument: Removes a document from a knowledge base
- queryKnowledgeBase: Queries a knowledge base for relevant information

Implementation:

```typescript
import { ApiClient, PaginatedResponse, PaginationParams } from './api';

export interface KnowledgeBase {
  id: string;
  name: string;
  description: string;
  organizationId: string;
  config?: KnowledgeBaseConfig;
  isPublic: boolean;
  createdAt: string;
  updatedAt: string;
}

export interface KnowledgeBaseConfig {
  chunkSize?: number;
  chunkOverlap?: number;
  embeddingModel?: string;
  vectorStore?: VectorStoreConfig;
  options?: Record<string, any>;
}

export interface VectorStoreConfig {
  provider: string;
  options?: Record<string, any>;
}

export interface Document {
  id: string;
  knowledgeBaseId: string;
  fileId: string;
  filename: string;
  status: DocumentStatus;
  chunkCount?: number;
  error?: string;
  createdAt: string;
  updatedAt: string;
}

export enum DocumentStatus {
  PENDING = 'pending',
  PROCESSING = 'processing',
  EMBEDDING = 'embedding',
  COMPLETED = 'completed',
  FAILED = 'failed',
}

export interface CreateKnowledgeBaseDto {
  name: string;
  description: string;
  organizationId: string;
  config?: KnowledgeBaseConfig;
  isPublic?: boolean;
}

export interface UpdateKnowledgeBaseDto {
  name?: string;
  description?: string;
  config?: KnowledgeBaseConfig;
  isPublic?: boolean;
}

export interface QueryKnowledgeBaseDto {
  query: string;
  limit?: number;
  minScore?: number;
}

export interface QueryResult {
  score: number;
  content: string;
  metadata: Record<string, any>;
}

export class KnowledgeBaseApi {
  private apiClient: ApiClient;

  constructor(apiClient: ApiClient) {
    this.apiClient = apiClient;
  }

  async getKnowledgeBases(params?: PaginationParams): Promise<PaginatedResponse<KnowledgeBase>> {
    return this.apiClient.get<PaginatedResponse<KnowledgeBase>>('/v1/knowledgeBase', params, { cache: true });
  }

  async getKnowledgeBase(id: string): Promise<KnowledgeBase> {
    return this.apiClient.get<KnowledgeBase>(`/v1/knowledgeBase/${id}`, undefined, { cache: true });
  }

  async createKnowledgeBase(createKnowledgeBaseDto: CreateKnowledgeBaseDto): Promise<KnowledgeBase> {
    const knowledgeBase = await this.apiClient.post<KnowledgeBase>('/v1/knowledgeBase', createKnowledgeBaseDto);
    this.apiClient.invalidateCache('/v1/knowledgeBase');
    return knowledgeBase;
  }

  async updateKnowledgeBase(id: string, updateKnowledgeBaseDto: UpdateKnowledgeBaseDto): Promise<KnowledgeBase> {
    const knowledgeBase = await this.apiClient.put<KnowledgeBase>(`/v1/knowledgeBase/${id}`, updateKnowledgeBaseDto);
    this.apiClient.invalidateCache('/v1/knowledgeBase');
    this.apiClient.invalidateCache(`/v1/knowledgeBase/${id}`);
    return knowledgeBase;
  }

  async deleteKnowledgeBase(id: string): Promise<void> {
    await this.apiClient.delete<void>(`/v1/knowledgeBase/${id}`);
    this.apiClient.invalidateCache('/v1/knowledgeBase');
    this.apiClient.invalidateCache(`/v1/knowledgeBase/${id}`);
  }

  async addDocuments(id: string, files: File[]): Promise<Document[]> {
    const formData = new FormData();
    files.forEach(file => formData.append('files', file));

    const response = await fetch(`${this.apiClient.getBaseUrl()}/v1/knowledgeBase/${id}/documents`, {
      method: 'POST',
      credentials: 'include',
      body: formData,
    });

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(errorData.message || 'Failed to add documents');
    }

    return response.json();
  }

  async listDocuments(id: string, params?: PaginationParams): Promise<PaginatedResponse<Document>> {
    return this.apiClient.get<PaginatedResponse<Document>>(`/v1/knowledgeBase/${id}/documents`, params, { cache: true });
  }

  async removeDocument(id: string, documentId: string): Promise<void> {
    await this.apiClient.delete<void>(`/v1/knowledgeBase/${id}/documents/${documentId}`);
    this.apiClient.invalidateCache(`/v1/knowledgeBase/${id}/documents`);
  }

  async queryKnowledgeBase(id: string, queryDto: QueryKnowledgeBaseDto): Promise<QueryResult[]> {
    return this.apiClient.post<QueryResult[]>(`/v1/knowledgeBase/${id}/query`, queryDto);
  }
}
```

1.3. File API Client
-----------------

The File API client provides methods for interacting with the file-related endpoints of the backend API.

File: apps/frontend/src/utils/fileApi.ts

The File API client extends the base API client and provides methods for uploading, retrieving, and deleting files.

Key Methods:
- getFiles: Retrieves a list of files with pagination and filtering
- getFile: Retrieves a single file by ID
- uploadFile: Uploads a new file
- deleteFile: Deletes a file
- downloadFile: Downloads a file
- getSupportedFileTypes: Gets supported file types

Implementation:

```typescript
import { ApiClient, PaginatedResponse, PaginationParams } from './api';

export interface File {
  id: string;
  organizationId: string;
  filename: string;
  mimetype: string;
  size: number;
  path: string;
  description?: string;
  processingStatus?: string;
  processingResult?: Record<string, any>;
  createdAt: string;
  updatedAt: string;
}

export interface FileType {
  mimetype: string;
  extension: string;
  description: string;
}

export interface UploadFileDto {
  organizationId: string;
  description?: string;
  processImmediately?: boolean;
}

export class FileApi {
  private apiClient: ApiClient;

  constructor(apiClient: ApiClient) {
    this.apiClient = apiClient;
  }

  async getFiles(params?: PaginationParams): Promise<PaginatedResponse<File>> {
    return this.apiClient.get<PaginatedResponse<File>>('/v1/file', params, { cache: true });
  }

  async getFile(id: string): Promise<File> {
    return this.apiClient.get<File>(`/v1/file/${id}`, undefined, { cache: true });
  }

  async uploadFile(file: globalThis.File, uploadFileDto: UploadFileDto): Promise<File> {
    const formData = new FormData();
    formData.append('file', file);
    formData.append('organizationId', uploadFileDto.organizationId);
    
    if (uploadFileDto.description) {
      formData.append('description', uploadFileDto.description);
    }
    
    if (uploadFileDto.processImmediately !== undefined) {
      formData.append('processImmediately', String(uploadFileDto.processImmediately));
    }

    const response = await fetch(`${this.apiClient.getBaseUrl()}/v1/file`, {
      method: 'POST',
      credentials: 'include',
      body: formData,
    });

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(errorData.message || 'Failed to upload file');
    }

    const uploadedFile = await response.json();
    this.apiClient.invalidateCache('/v1/file');
    return uploadedFile;
  }

  async deleteFile(id: string): Promise<void> {
    await this.apiClient.delete<void>(`/v1/file/${id}`);
    this.apiClient.invalidateCache('/v1/file');
    this.apiClient.invalidateCache(`/v1/file/${id}`);
  }

  async downloadFile(id: string): Promise<Blob> {
    const response = await fetch(`${this.apiClient.getBaseUrl()}/v1/file/${id}/download`, {
      method: 'GET',
      credentials: 'include',
    });

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(errorData.message || 'Failed to download file');
    }

    return response.blob();
  }

  async getSupportedFileTypes(): Promise<FileType[]> {
    return this.apiClient.get<FileType[]>('/v1/file/types', undefined, { cache: true });
  }
}
```

2. CONTEXT PROVIDERS
==============================================================

Context providers in Jibu Console are React components that provide global state and functionality to the application. They use React's Context API to make data and functions available to all components in the component tree without having to pass props down manually.

2.1. API Context Provider
-----------------------

The API Context Provider provides access to the API clients throughout the application.

File: apps/frontend/src/utils/apiContext.tsx

The API Context Provider initializes the API clients and makes them available to all components in the application. It also handles authentication and provides methods for managing the API clients.

Key Features:
- Provides access to all API clients
- Handles authentication token management
- Provides methods for refreshing API clients
- Manages API client configuration

Implementation:

```typescript
import React, { createContext, useContext, useEffect, useState } from 'react';
import { ApiClient } from './api';
import { AgentApi } from './AgentApi';
import { AssistantsApi } from './AssistantsApi';
import { ChatApi } from './chatApi';
import { KnowledgeBaseApi } from './knowledgebaseApi';
import { FileApi } from './fileApi';
import { useSupabase } from './supabase/client';

interface ApiContextValue {
  apiClient: ApiClient;
  agentApi: AgentApi;
  assistantsApi: AssistantsApi;
  chatApi: ChatApi;
  knowledgeBaseApi: KnowledgeBaseApi;
  fileApi: FileApi;
  refreshApis: () => void;
  isInitialized: boolean;
}

const ApiContext = createContext<ApiContextValue | undefined>(undefined);

export const ApiProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const { supabase, session } = useSupabase();
  const [apiClient, setApiClient] = useState<ApiClient | null>(null);
  const [agentApi, setAgentApi] = useState<AgentApi | null>(null);
  const [assistantsApi, setAssistantsApi] = useState<AssistantsApi | null>(null);
  const [chatApi, setChatApi] = useState<ChatApi | null>(null);
  const [knowledgeBaseApi, setKnowledgeBaseApi] = useState<KnowledgeBaseApi | null>(null);
  const [fileApi, setFileApi] = useState<FileApi | null>(null);
  const [isInitialized, setIsInitialized] = useState(false);

  const initializeApis = () => {
    const headers: Record<string, string> = {
      'Content-Type': 'application/json',
    };

    // Add authentication token if available
    if (session?.access_token) {
      headers['Authorization'] = `Bearer ${session.access_token}`;
    }

    // Initialize API client
    const baseUrl = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:3000/api';
    const client = new ApiClient(baseUrl, headers);

    // Initialize API clients
    const agent = new AgentApi(client);
    const assistants = new AssistantsApi(client);
    const chat = new ChatApi(client);
    const knowledgeBase = new KnowledgeBaseApi(client);
    const file = new FileApi(client);

    // Set state
    setApiClient(client);
    setAgentApi(agent);
    setAssistantsApi(assistants);
    setChatApi(chat);
    setKnowledgeBaseApi(knowledgeBase);
    setFileApi(file);
    setIsInitialized(true);
  };

  const refreshApis = () => {
    initializeApis();
  };

  // Initialize APIs when session changes
  useEffect(() => {
    initializeApis();
  }, [session]);

  // Return provider with value
  return (
    <ApiContext.Provider
      value={{
        apiClient: apiClient!,
        agentApi: agentApi!,
        assistantsApi: assistantsApi!,
        chatApi: chatApi!,
        knowledgeBaseApi: knowledgeBaseApi!,
        fileApi: fileApi!,
        refreshApis,
        isInitialized,
      }}
    >
      {children}
    </ApiContext.Provider>
  );
};

export const useApi = (): ApiContextValue => {
  const context = useContext(ApiContext);
  if (!context) {
    throw new Error('useApi must be used within an ApiProvider');
  }
  return context;
};
```

2.2. Organization Context Provider
--------------------------------

The Organization Context Provider provides access to the current organization and organization-related functionality throughout the application.

File: apps/frontend/src/utils/organizationContext.tsx

The Organization Context Provider manages the current organization and provides methods for switching between organizations. It also handles organization-related data and functionality.

Key Features:
- Provides access to the current organization
- Provides methods for switching between organizations
- Manages organization-related data and functionality
- Integrates with the API Context Provider

Implementation:

```typescript
import React, { createContext, useContext, useEffect, useState } from 'react';
import { useApi } from './apiContext';
import { useSupabase } from './supabase/client';

export interface Organization {
  id: string;
  name: string;
  description?: string;
  settings?: OrganizationSettings;
  createdAt: string;
  updatedAt: string;
}

export interface OrganizationSettings {
  defaultAgentProvider?: string;
  defaultAgentModel?: string;
  defaultVectorStore?: string;
  defaultEmbeddingModel?: string;
  allowedDomains?: string[];
  maxAssistants?: number;
  maxKnowledgeBases?: number;
  maxAgents?: number;
  options?: Record<string, any>;
}

interface OrganizationContextValue {
  currentOrganization: Organization | null;
  organizations: Organization[];
  isLoading: boolean;
  error: string | null;
  switchOrganization: (organizationId: string) => void;
  refreshOrganizations: () => void;
}

const OrganizationContext = createContext<OrganizationContextValue | undefined>(undefined);

export const OrganizationProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const { apiClient, isInitialized } = useApi();
  const { session } = useSupabase();
  const [currentOrganization, setCurrentOrganization] = useState<Organization | null>(null);
  const [organizations, setOrganizations] = useState<Organization[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchOrganizations = async () => {
    if (!isInitialized || !session) {
      return;
    }

    setIsLoading(true);
    setError(null);

    try {
      const response = await apiClient.get<{ items: Organization[] }>('/v1/organization', {
        limit: 100,
      });

      setOrganizations(response.items);

      // Set current organization from local storage or first organization
      const storedOrgId = localStorage.getItem('currentOrganizationId');
      const org = storedOrgId
        ? response.items.find(o => o.id === storedOrgId) || response.items[0]
        : response.items[0];

      if (org) {
        setCurrentOrganization(org);
        localStorage.setItem('currentOrganizationId', org.id);
      }
    } catch (err) {
      setError('Failed to fetch organizations');
      console.error('Failed to fetch organizations:', err);
    } finally {
      setIsLoading(false);
    }
  };

  const switchOrganization = (organizationId: string) => {
    const org = organizations.find(o => o.id === organizationId);
    if (org) {
      setCurrentOrganization(org);
      localStorage.setItem('currentOrganizationId', org.id);
    }
  };

  const refreshOrganizations = () => {
    fetchOrganizations();
  };

  // Fetch organizations when session or API client changes
  useEffect(() => {
    fetchOrganizations();
  }, [isInitialized, session]);

  // Return provider with value
  return (
    <OrganizationContext.Provider
      value={{
        currentOrganization,
        organizations,
        isLoading,
        error,
        switchOrganization,
        refreshOrganizations,
      }}
    >
      {children}
    </OrganizationContext.Provider>
  );
};

export const useOrganization = (): OrganizationContextValue => {
  const context = useContext(OrganizationContext);
  if (!context) {
    throw new Error('useOrganization must be used within an OrganizationProvider');
  }
  return context;
};
```
