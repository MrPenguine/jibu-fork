1.1. Console Service
------------------

The Console Service manages console-specific operations, such as retrieving system statistics and performing administrative tasks.

File: apps/backend/src/modules/v1/console/console.service.ts

Key Methods:
- getSystemStats(): Retrieves system statistics
- checkSystemHealth(): Checks system health
- resetSystem(): Resets system to default state (admin only)
- getSystemLogs(options): Retrieves system logs with filtering and pagination
- getUsageMetrics(organizationId): Retrieves usage metrics for an organization

Implementation Details:
- Integrates with various services to collect system-wide statistics
- Implements health checks for all system components
- Provides administrative functions for system management
- Collects and aggregates usage metrics
- Implements logging and monitoring capabilities

Code Example (Simplified):
```typescript
@Injectable()
export class ConsoleService {
  constructor(
    private readonly userService: UserService,
    private readonly organizationService: OrganizationService,
    private readonly assistantService: AssistantService,
    private readonly agentService: AgentService,
    private readonly knowledgeBaseService: KnowledgeBaseService,
    private readonly chatService: ChatsService,
    @InjectRedis() private readonly redis: Redis,
  ) {}

  async getSystemStats(): Promise<SystemStats> {
    const [
      userCount,
      organizationCount,
      assistantCount,
      agentCount,
      knowledgeBaseCount,
      chatCount,
    ] = await Promise.all([
      this.userService.count(),
      this.organizationService.count(),
      this.assistantService.count(),
      this.agentService.count(),
      this.knowledgeBaseService.count(),
      this.chatService.count(),
    ]);

    return {
      userCount,
      organizationCount,
      assistantCount,
      agentCount,
      knowledgeBaseCount,
      chatCount,
      systemUptime: process.uptime(),
      timestamp: new Date().toISOString(),
    };
  }

  async checkSystemHealth(): Promise<SystemHealth> {
    const checks = await Promise.all([
      this.checkDatabaseHealth(),
      this.checkRedisHealth(),
      this.checkExternalServicesHealth(),
    ]);

    const allHealthy = checks.every(check => check.status === 'healthy');

    return {
      status: allHealthy ? 'healthy' : 'unhealthy',
      checks: {
        database: checks[0],
        redis: checks[1],
        externalServices: checks[2],
      },
      timestamp: new Date().toISOString(),
    };
  }

  private async checkDatabaseHealth(): Promise<HealthCheckResult> {
    try {
      // Perform a simple query to check database connectivity
      await this.userService.count();
      return { status: 'healthy', message: 'Database connection is healthy' };
    } catch (error) {
      return {
        status: 'unhealthy',
        message: 'Database connection failed',
        error: error.message,
      };
    }
  }

  private async checkRedisHealth(): Promise<HealthCheckResult> {
    try {
      // Perform a simple Redis operation to check connectivity
      await this.redis.ping();
      return { status: 'healthy', message: 'Redis connection is healthy' };
    } catch (error) {
      return {
        status: 'unhealthy',
        message: 'Redis connection failed',
        error: error.message,
      };
    }
  }
}
```

```typescript
@Injectable()
export class CredentialService {
  constructor(
    @InjectRepository(CredentialEntity)
    private readonly credentialRepository: Repository<CredentialEntity>,
    private readonly encryptionService: EncryptionService,
    private readonly credentialValidatorFactory: CredentialValidatorFactory,
  ) {}

  async createCredential(createCredentialDto: CreateCredentialDto): Promise<Credential> {
    // Encrypt sensitive data
    const encryptedData = await this.encryptionService.encrypt(
      JSON.stringify(createCredentialDto.data),
    );

    // Create credential entity
    const credential = this.credentialRepository.create({
      ...createCredentialDto,
      data: encryptedData,
    });

    // Save credential
    return this.credentialRepository.save(credential);
  }

  async findOne(id: string): Promise<Credential> {
    const credential = await this.credentialRepository.findOne({ where: { id } });
    
    if (!credential) {
      throw new NotFoundException(`Credential with ID ${id} not found`);
    }

    // Decrypt sensitive data
    const decryptedData = JSON.parse(
      await this.encryptionService.decrypt(credential.data),
    );

    return {
      ...credential,
      data: decryptedData,
    };
  }

  async validateCredential(id: string): Promise<ValidationResult> {
    const credential = await this.findOne(id);
    
    // Get validator for credential type
    const validator = this.credentialValidatorFactory.getValidator(credential.type);
    
    if (!validator) {
      throw new BadRequestException(`No validator found for credential type ${credential.type}`);
    }

    // Validate credential
    return validator.validate(credential.data);
  }

  getCredentialTypes(): CredentialType[] {
```

```typescript
@Injectable()
export class FileService {
  constructor(
    @InjectRepository(FileEntity)
    private readonly fileRepository: Repository<FileEntity>,
    private readonly storageService: StorageService,
    private readonly fileProcessorFactory: FileProcessorFactory,
  ) {}

  async uploadFile(file: Express.Multer.File, uploadFileDto: UploadFileDto): Promise<File> {
    // Validate file type
    this.validateFileType(file.mimetype);

    // Upload file to storage
    const storagePath = await this.storageService.uploadFile(
      file.buffer,
      file.originalname,
      file.mimetype,
    );

    // Create file entity
    const fileEntity = this.fileRepository.create({
      ...uploadFileDto,
      filename: file.originalname,
      mimetype: file.mimetype,
      size: file.size,
      path: storagePath,
    });

    // Save file entity
    const savedFile = await this.fileRepository.save(fileEntity);

    // Process file if needed
    if (uploadFileDto.processImmediately) {
      await this.processFile(savedFile.id);
    }

    return savedFile;
  }

  async downloadFile(id: string): Promise<{ buffer: Buffer; mimetype: string; filename: string }> {
    const file = await this.findOne(id);
    
    // Download file from storage
    const buffer = await this.storageService.downloadFile(file.path);
    
    return {
      buffer,
      mimetype: file.mimetype,
      filename: file.filename,
    };
  }

  async processFile(id: string): Promise<ProcessingResult> {
    const file = await this.findOne(id);
    
    // Get processor for file type
    const processor = this.fileProcessorFactory.getProcessor(file.mimetype);
    
    if (!processor) {
      throw new BadRequestException(`No processor found for file type ${file.mimetype}`);
    }

    // Download file from storage
    const buffer = await this.storageService.downloadFile(file.path);
    
    // Process file
    const result = await processor.process(buffer, file);
    
    // Update file with processing results
    file.processingStatus = 'completed';
    file.processingResult = result;
    await this.fileRepository.save(file);
    
    return result;
  }

  getSupportedFileTypes(): FileType[] {
    return [
      {
        mimetype: 'application/pdf',
        extension: 'pdf',
        description: 'PDF Document',
      },
      {
        mimetype: 'text/plain',
        extension: 'txt',
        description: 'Text File',
      },
      {
        mimetype: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
        extension: 'docx',
        description: 'Word Document',
      },
      // Other supported file types...
    ];
  }
}
```

Code Example (Simplified):
```typescript
@Injectable()
export class KnowledgeBaseService {
  constructor(
    @InjectRepository(KnowledgeBaseEntity)
    private readonly knowledgeBaseRepository: Repository<KnowledgeBaseEntity>,
    @InjectRepository(DocumentEntity)
    private readonly documentRepository: Repository<DocumentEntity>,
    private readonly fileService: FileService,
    private readonly vectorDbService: VectorDbService,
    private readonly textSplitterService: TextSplitterService,
    private readonly embeddingService: EmbeddingService,
  ) {}

  async createKnowledgeBase(createKnowledgeBaseDto: CreateKnowledgeBaseDto): Promise<KnowledgeBase> {
    // Create knowledge base entity
    const knowledgeBase = this.knowledgeBaseRepository.create(createKnowledgeBaseDto);
    
    // Save knowledge base
    return this.knowledgeBaseRepository.save(knowledgeBase);
  }

  async addDocuments(id: string, files: Express.Multer.File[]): Promise<Document[]> {
    const knowledgeBase = await this.findOne(id);
    const documents: Document[] = [];

    for (const file of files) {
      // Upload file
      const uploadedFile = await this.fileService.uploadFile(file, {
        organizationId: knowledgeBase.organizationId,
        description: `Document for knowledge base ${knowledgeBase.name}`,
        processImmediately: false,
      });

      // Process file content
      const { buffer, mimetype } = await this.fileService.downloadFile(uploadedFile.id);
      const textContent = await this.extractTextFromFile(buffer, mimetype);
      
      // Split text into chunks
      const chunks = await this.textSplitterService.splitText(textContent, {
        chunkSize: knowledgeBase.chunkSize || 1000,
        chunkOverlap: knowledgeBase.chunkOverlap || 200,
      });

      // Create document entity
      const document = this.documentRepository.create({
        knowledgeBaseId: id,
        fileId: uploadedFile.id,
        filename: uploadedFile.filename,
        status: 'processing',
      });
      
      // Save document
      const savedDocument = await this.documentRepository.save(document);
      documents.push(savedDocument);

      // Process chunks and create embeddings (async)
      this.processDocumentChunks(savedDocument, chunks).catch(error => {
        console.error(`Error processing document chunks: ${error.message}`, error.stack);
      });
    }

    return documents;
  }

  async queryKnowledgeBase(id: string, query: string, options?: QueryOptions): Promise<QueryResult> {
    const knowledgeBase = await this.findOne(id);
    
    // Generate query embedding
    const queryEmbedding = await this.embeddingService.generateEmbedding(query);
    
    // Query vector database
    const results = await this.vectorDbService.query({
      collectionName: `kb_${id}`,
      queryEmbedding,
      limit: options?.limit || 5,
      minScore: options?.minScore || 0.7,
    });

    return {
      query,
      results: results.map(result => ({
        score: result.score,
        content: result.document.content,
        metadata: result.document.metadata,
      })),
    };
  }

  private async processDocumentChunks(document: Document, chunks: string[]): Promise<void> {
    try {
      // Update document status
      document.status = 'embedding';
      await this.documentRepository.save(document);

      // Generate embeddings for chunks
      const embeddings = await Promise.all(
        chunks.map(chunk => this.embeddingService.generateEmbedding(chunk)),
      );

      // Store chunks and embeddings in vector database
      await this.vectorDbService.addDocuments({
        collectionName: `kb_${document.knowledgeBaseId}`,
        documents: chunks.map((chunk, index) => ({
          content: chunk,
          embedding: embeddings[index],
          metadata: {
            documentId: document.id,
            filename: document.filename,
            chunkIndex: index,
          },
        })),
      });

      // Update document status
      document.status = 'completed';
      document.chunkCount = chunks.length;
      await this.documentRepository.save(document);
    } catch (error) {
      // Update document status on error
      document.status = 'failed';
      document.error = error.message;
      await this.documentRepository.save(document);
      throw error;
    }
  }
}
```

```typescript
@Injectable()
export class OrganizationService {
  constructor(
    @InjectRepository(OrganizationEntity)
    private readonly organizationRepository: Repository<OrganizationEntity>,
    @InjectRepository(OrganizationMemberEntity)
    private readonly organizationMemberRepository: Repository<OrganizationMemberEntity>,
    private readonly userService: UserService,
  ) {}

  async createOrganization(createOrganizationDto: CreateOrganizationDto): Promise<Organization> {
    // Create organization entity
    const organization = this.organizationRepository.create(createOrganizationDto);
    
    // Save organization
    const savedOrganization = await this.organizationRepository.save(organization);

    // Add creator as admin member
    if (createOrganizationDto.creatorId) {
      await this.addMember(savedOrganization.id, createOrganizationDto.creatorId, 'admin');
    }

    return savedOrganization;
  }

  async addMembers(id: string, memberData: { userId: string; role: MemberRole }[]): Promise<void> {
    const organization = await this.findOne(id);

    for (const { userId, role } of memberData) {
      // Validate user
      await this.userService.findOne(userId);

      // Check if user is already a member
      const existingMember = await this.organizationMemberRepository.findOne({
        where: { organizationId: id, userId },
      });

      if (existingMember) {
        // Update role if different
        if (existingMember.role !== role) {
          existingMember.role = role;
          await this.organizationMemberRepository.save(existingMember);
        }
      } else {
        // Add new member
        await this.addMember(id, userId, role);
      }
    }
  }

  async listMembers(id: string, options?: PaginationQueryDto): Promise<PaginatedResponse<OrganizationMember>> {
    const organization = await this.findOne(id);

    const [members, totalItems] = await this.organizationMemberRepository.findAndCount({
      where: { organizationId: id },
      skip: options ? (options.page - 1) * options.limit : 0,
      take: options?.limit || 10,
      relations: ['user'],
    });

    return {
      items: members.map(member => ({
        userId: member.userId,
        organizationId: member.organizationId,
        role: member.role,
        joinedAt: member.createdAt,
        user: member.user,
      })),
      meta: {
        totalItems,
        itemCount: members.length,
        itemsPerPage: options?.limit || 10,
        totalPages: Math.ceil(totalItems / (options?.limit || 10)),
        currentPage: options?.page || 1,
      },
    };
  }

  private async addMember(organizationId: string, userId: string, role: MemberRole): Promise<OrganizationMember> {
    const member = this.organizationMemberRepository.create({
      organizationId,
      userId,
      role,
    });

    await this.organizationMemberRepository.save(member);

    return {
      userId,
      organizationId,
      role,
      joinedAt: member.createdAt,
    };
  }
}
```

```typescript
@Injectable()
export class UserService {
  constructor(
    @InjectRepository(UserEntity)
    private readonly userRepository: Repository<UserEntity>,
    private readonly supabaseService: SupabaseService,
    private readonly jwtService: JwtService,
  ) {}

  async register(registrationData: RegisterDto): Promise<{ user: User; token: string }> {
    // Check if user already exists
    const existingUser = await this.userRepository.findOne({
      where: { email: registrationData.email },
    });

    if (existingUser) {
      throw new ConflictException('User with this email already exists');
    }

    // Register user with Supabase
    const { data: authData, error } = await this.supabaseService.auth.signUp({
      email: registrationData.email,
      password: registrationData.password,
    });

    if (error) {
      throw new BadRequestException(error.message);
    }

    // Create user entity
    const user = this.userRepository.create({
      id: authData.user.id,
      email: registrationData.email,
      name: registrationData.name,
      role: 'user',
    });

    // Save user
    const savedUser = await this.userRepository.save(user);

    // Generate JWT token
    const token = this.jwtService.sign({
      sub: savedUser.id,
      email: savedUser.email,
      role: savedUser.role,
    });

    return { user: savedUser, token };
  }

  async login(credentials: LoginDto): Promise<{ user: User; token: string }> {
    // Authenticate with Supabase
    const { data: authData, error } = await this.supabaseService.auth.signInWithPassword({
      email: credentials.email,
      password: credentials.password,
    });

    if (error) {
      throw new UnauthorizedException('Invalid credentials');
    }

    // Get user from database
    const user = await this.userRepository.findOne({
      where: { id: authData.user.id },
    });

    if (!user) {
      throw new NotFoundException('User not found');
    }

    // Generate JWT token
    const token = this.jwtService.sign({
      sub: user.id,
      email: user.email,
      role: user.role,
    });

    return { user, token };
  }

  async getProfile(userId: string): Promise<User> {
    const user = await this.findOne(userId);
    
    // Get user organizations
    const organizations = await this.organizationService.getUserOrganizations(userId);
    
    return {
      ...user,
      organizations,
    };
  }
}
```

```typescript
@Injectable()
export class LangflowChatHistoryService {
  private readonly logger = new Logger(LangflowChatHistoryService.name);
  private readonly localCache: Map<string, { messages: ChatMessage[]; timestamp: number }> = new Map();
  private readonly cacheTTL = 5000; // 5 seconds

  constructor(@InjectRedis() private readonly redis: Redis) {}

  async getChatHistory(chatId: string): Promise<ChatMessage[]> {
    // Check local cache first
    const cacheKey = `chat:history:${chatId}`;
    const cachedData = this.localCache.get(cacheKey);
    
    if (cachedData && Date.now() - cachedData.timestamp < this.cacheTTL) {
      return cachedData.messages;
    }

    // Get messages from Redis
    const messagesJson = await this.redis.lrange(`chat:history:${chatId}`, 0, -1);
    
    if (!messagesJson.length) {
      return [];
    }

    // Parse messages
    const messages = messagesJson.map(json => JSON.parse(json) as ChatMessage);
    
    // Update local cache
    this.localCache.set(cacheKey, {
      messages,
      timestamp: Date.now(),
    });

    return messages;
  }

  async addMessage(chatId: string, message: ChatMessage): Promise<void> {
    // Store in Redis
    await this.redis.rpush(`chat:history:${chatId}`, JSON.stringify(message));
    
    // Update local cache
    const cacheKey = `chat:history:${chatId}`;
    const cachedData = this.localCache.get(cacheKey);
    
    if (cachedData) {
      cachedData.messages.push(message);
      cachedData.timestamp = Date.now();
    }
  }

  async getChatHistoryByAssistantAndSession(assistantId: string, sessionId: string): Promise<ChatMessage[]> {
    // Get messages from Redis using the backward compatibility key pattern
    const messagesJson = await this.redis.lrange(`chat:${assistantId}:${sessionId}`, 0, -1);
    
    if (!messagesJson.length) {
      return [];
    }

    // Parse messages
    return messagesJson.map(json => JSON.parse(json) as ChatMessage);
  }
}
```

- Chat history storage and retrieval
- Caching of frequently accessed data
- Real-time message streaming
- Rate limiting for API endpoints
- Session management for authentication

The Redis-based chat history system is a key component of the Jibu Console, providing efficient storage and retrieval of chat messages for assistants. It implements a dual key pattern for backward compatibility and includes a local caching mechanism for improved performance.
