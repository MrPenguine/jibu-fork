### 2. Assistant Integration

Enhanced assistant integration within workflow:

```typescript
// Assistant node processor
const assistantNodeProcessor: NodeProcessor = async (node, session, executionService) => {
  const { assistantId, promptTemplate, systemMessage, tools } = node.data;
  
  if (!assistantId) {
    throw new BadRequestException('Assistant ID is required for assistant node');
  }
  
  // Resolve variables in prompt template
  const prompt = await executionService.resolveVariables(
    promptTemplate, 
    session.variables as Record<string, any>
  );
  
  // Get conversation history from session
  const conversationHistory = session.history?.messages || [];
  
  // Call assistant with context
  const assistantResponse = await executionService.assistantsService.sendMessage({
    assistantId,
    message: prompt,
    systemMessage,
    history: conversationHistory,
    tools,
  });
  
  // Store response in variables
  const updatedVariables = {
    ...session.variables as Record<string, any>,
    lastAssistantResponse: assistantResponse.content,
  };
  
  // Update session variables
  await executionService.prisma.agentSession.update({
    where: { id: session.id },
    data: {
      variables: updatedVariables,
      history: {
        messages: [
          ...(session.history?.messages || []),
          { role: 'user', content: prompt },
          { role: 'assistant', content: assistantResponse.content }
        ]
      }
    },
  });
  
  return {
    response: assistantResponse.content,
    waitForUserInput: false,
  };
};
```
